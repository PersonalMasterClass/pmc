    convert the days to a readable string of day names.

def list_days
        d = ["Mondays,", "Tuesdays,", "Wednesdays,", "Thursdays,", "Fridays,", "Saturdays,", "Sundays,"]
        x = []

        boolDays = get_days

        for i in 0..d.length
                if boolDays[i] == true
                        x << d[i]
                end
        end
        x.insert(x.length - 1, "and")
        return (x*" ").chop
end

def to_string(time)
        begin
                x= (time/60).floor.to_s.rjust(2,'0') + ":" + (time % 60).floor.to_s.rjust(2,'0')    
        rescue
                x = time
        end
        return x;
end
